        function updateTranslations() {
            const dict = i18n[settings.lang];
            document.querySelectorAll('[data-key]').forEach(el => {
                const key = el.getAttribute('data-key');
                if(dict[key]) el.innerText = dict[key];
            });
            document.querySelectorAll('[data-placeholder-key]').forEach(el => {
                const key = el.getAttribute('data-placeholder-key');
                if(dict[key]) el.placeholder = dict[key];
            });
        }

        function toggleDarkMode() {
            settings.darkMode = !settings.darkMode;
            document.body.classList.toggle('dark', settings.darkMode);
            localStorage.setItem('mm_mobile_glass_settings', JSON.stringify(settings));
        }

        function changeLanguage() {
            settings.lang = document.getElementById('langSelect').value;
            localStorage.setItem('mm_mobile_glass_settings', JSON.stringify(settings));
            updateTranslations();
        }

            settings.scriptUrl = document.getElementById('scriptUrl').value;
            localStorage.setItem('mm_mobile_glass_settings', JSON.stringify(settings));

            showToast(i18n[settings.lang]['toast-saved']);
        }

        function toggleModal(id) {
            document.getElementById(id).classList.toggle('hidden');
        }

        function switchTab(tab) {
            const isInv = tab === 'inventory';
            document.getElementById('view-inventory').classList.toggle('hidden', !isInv);
            document.getElementById('view-sales').classList.toggle('hidden', isInv);
            document.getElementById('tab-inventory').className = `flex-1 py-3 rounded-xl text-xs font-bold transition-all ${isInv ? 'tab-active' : 'text-slate-500'}`;
            document.getElementById('tab-sales').className = `flex-1 py-3 rounded-xl text-xs font-bold transition-all ${!isInv ? 'tab-active' : 'text-slate-500'}`;
        }

        function formatMMK(num) {
            return new Intl.NumberFormat('en-MM').format(num) + ' Ks';
        }

        async function syncToGoogleSheets(action, data) {
            if (!settings.scriptUrl) return;
            document.getElementById('syncIndicator').classList.remove('hidden');
            try {
                await fetch(settings.scriptUrl, {
                    method: 'POST',
                    mode: 'no-cors',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ action, ...data, timestamp: new Date().toLocaleString() })
                });
            } catch (err) { console.error(err); }
            finally { setTimeout(() => document.getElementById('syncIndicator').classList.add('hidden'), 1000); }
        }

                category: item.category

            document.getElementById('sellPreview').innerHTML = `
                <div class="flex justify-between items-center mb-1">
                    <span class="font-black text-indigo-900 dark:text-indigo-100">${item.model}</span>
                </div>
                <p class="text-[10px] text-indigo-400 font-mono mb-3">IMEI: ${item.imei}</p>
                <div class="flex justify-between text-xs font-bold">
                    <span>Listing Price:</span>
                    <span class="text-indigo-600">${formatMMK(item.salePrice)}</span>
                </div>
            `;
            document.getElementById('finalSoldPrice').value = item.salePrice;
            toggleModal('sellDeviceModal');
        }

        document.getElementById('confirmSaleBtn').addEventListener('click', async () => {
            const soldPrice = parseFloat(document.getElementById('finalSoldPrice').value);
            const index = inventory.findIndex(i => i.id === currentSaleId);
            if (index > -1) {
                const item = inventory[index];
                const saleRecord = { ...item, soldPrice, profit: soldPrice - item.purchasePrice, dateSold: new Date().toISOString() };
                sales.push(saleRecord);
                inventory.splice(index, 1);
                saveData();
                renderInventory();
                renderSales();
                toggleModal('sellDeviceModal');
                await syncToGoogleSheets('RECORD_SALE', {
                    item_model: saleRecord.model,
                    imei: saleRecord.imei,
                    sold_at: saleRecord.soldPrice,
                    profit: saleRecord.profit
                });
                showToast(i18n[settings.lang]['toast-sold']);
            }
        });

        function saveData() {
            localStorage.setItem('mm_mobile_glass_inv', JSON.stringify(inventory));
            localStorage.setItem('mm_mobile_glass_sales', JSON.stringify(sales));
            updateStats();
        }

        function updateStats() {
            document.getElementById('totalStockCount').innerText = inventory.length;
            document.getElementById('itemCountLabel').innerText = `${inventory.length} Items`;
            const today = new Date().toLocaleDateString();
            const incomeToday = sales.filter(s => new Date(s.dateSold).toLocaleDateString() === today).reduce((sum, s) => sum + s.soldPrice, 0);
            document.getElementById('todayIncome').innerText = formatMMK(incomeToday);
        }

        function renderInventory() {
            const container = document.getElementById('inventoryContainer');
            const search = document.getElementById('inventorySearch').value.toLowerCase();
            container.innerHTML = '';
            inventory.filter(i => i.model.toLowerCase().includes(search) || i.imei.toLowerCase().includes(search)).forEach(item => {
                const div = document.createElement('div');
                div.className = 'glass-card p-5 rounded-3xl flex justify-between items-center';
                div.innerHTML = `
                    <div class="space-y-1">
                        <span class="text-[8px] bg-slate-100 dark:bg-slate-800 px-2 py-0.5 rounded-full font-black uppercase">${item.category}</span>
                        <h4 class="font-black">${item.model}</h4>
                        <p class="text-[9px] text-slate-400 font-mono">${item.imei}</p>
                        <p class="text-sm font-black text-indigo-600 mt-1">${formatMMK(item.salePrice)}</p>
                    </div>
                    <div class="flex gap-2">
                        <button onclick="prepareSale(${item.id})" class="h-10 w-10 bg-emerald-500/10 text-emerald-600 rounded-xl flex items-center justify-center"><i class="fas fa-cart-shopping"></i></button>
                        <button onclick="deleteItem(${item.id})" class="h-10 w-10 bg-red-500/10 text-red-500 rounded-xl flex items-center justify-center"><i class="fas fa-trash-can"></i></button>
                    </div>
                `;
                container.appendChild(div);
            });
        }

        function renderSales() {
            const container = document.getElementById('salesContainer');
            container.innerHTML = '';
            sales.sort((a,b) => new Date(b.dateSold) - new Date(a.dateSold)).forEach(sale => {
                const div = document.createElement('div');
                div.className = 'glass-card p-5 rounded-3xl border-l-4 border-emerald-500 flex justify-between items-center';
                div.innerHTML = `
                    <div><h4 class="font-black text-sm">${sale.model}</h4><p class="text-[9px] text-slate-400">${new Date(sale.dateSold).toLocaleString()}</p></div>
                    <div class="text-right"><p class="font-black text-emerald-600">${formatMMK(sale.soldPrice)}</p></div>
                `;
                container.appendChild(div);
            });
        }

        function deleteItem(id) { if (confirm("Remove item?")) { inventory = inventory.filter(i => i.id !== id); saveData(); renderInventory(); } }

        function showToast(msg) {
            const t = document.createElement('div');
            t.className = 'fixed top-20 left-1/2 -translate-x-1/2 bg-slate-900 text-white px-6 py-2 rounded-xl text-[10px] z-[100]';
            t.innerText = msg;
            document.body.appendChild(t);
            setTimeout(() => t.remove(), 2000);
        }

        document.getElementById('inventorySearch').addEventListener('input', renderInventory);
        window.onload = initApp;
    </script>
